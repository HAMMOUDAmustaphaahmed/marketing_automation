  users                 → id, name, email, password, company_id, role
  companies             → id, name, logo, country, sector, description, keywords, icp_profile
  competitors           → id, company_id, name, website, country, similarity_score, swot_analysis, ...
  prospects             → id, company_id, company_name, sector, email, phone, relevance_score, status, ...

  quote_items           → id, quote_id, description, quantity, unit_price, discount, total
  orders                → id, opportunity_id, order_number, amount, delivery_status, pv_file_path

  activity_logs         → id, company_id, user_id, entity_type, action, description

TECHNOLOGIES UTILISÉES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Backend       : Python 3.10+ / Flask 3.0
  Base de données : MySQL 8.0 / SQLAlchemy ORM
  IA            : Groq API (xAI) — modèle groq-3
  PDF           : ReportLab
  Email         : Flask-Mail (SMTP)
  Auth          : Flask-Login + Werkzeug (hash bcrypt)
  Frontend      : Bootstrap 5.3 + Chart.js + Font Awesome 6
  Upload        : Werkzeug secure_filename
